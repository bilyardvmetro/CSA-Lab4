.data
in_addr:    .word 0x01
out_addr:   .word 0x02
question:   .byte 18, 'What is your name'
buf:        .byte 7, 'Hello, '

; s1 - in ptr
; s2 - out ptr
; s3 - question ptr
; s4 - buf str ptr
; s5 - buf ptr
; s6 - buf len
; s7 - name counter
; s8 - name char

.code
    ; load in ptr
	lui s1, %hi(in_addr)
	ori s1, s1, %lo(in_addr)
	lw s1, s1

	; load out ptr
	lui s2, %hi(out_addr)
	ori s2, s2, %lo(out_addr)
	lw s2, s2

	; load question ptr
	lui s3, %hi(question)
    ori s3, s3, %lo(question)

    ; load buf ptr
    lui s4, %hi(buf)
    ori s4, s4, %lo(buf)

    addi s5, s4, 0      ; s4 -> s5

    lb s6, s4           ; 7 -> s6
    addi s4, s4, 1      ; s4++
    add s4, s4, s6      ; s4 += 7

    lw s7, s1

get_name:
    addi s7, s7, -1
    beq s7, r0, output

    lw s8, s1
    sb s8, s4
    addi s6, s6, 1
    sb s5, s6

    jal r0, get_name

output:
    addi a0, s4, 0
    addi a1, s2, 0
    jal ra, print_pstr

    addi a0, s5, 0
    addi a1, s2, 0
    jal ra, print_pstr

    jal r0, end

; t0 - str ptr
; t1 - str len
; t2 - str char
; t6 - out ptr

print_pstr:
    addi t0, a0, 0
    addi t6, a1, 0

    lw t1, t0
    addi t0, t0, 1

loop:
    addi t1, t1, -1
    beq t1, r0, ret

    lb t2, t0
    sw t2, t6

    jal r0, loop

ret:
    jalr r0, ra, 0

end:
	halt
