.data
    in_addr:    0
    out_addr:   1
    buf:        11, "hello world"

.code
    lui bp, %hi(in_addr)
    ori bp, bp, %lo(in_addr)
    lw bp, bp, 0

    lui s2, %hi(out_addr)
    ori s2, s2, %lo(out_addr)
    lw s2, s2, 0

    lw s3, bp, 0
    addi bp, bp, 1
	
loop:
	addi s3, s3, -1			; s3-1 -> s3
	beq s3, r0, end			; if s3 == 0 go to end
	
	lw s4, bp, 0			; mem[bp] -> s4. Символ в s4
	sw s2, s4, 0			; s4 -> mem[s2]. Вывести символ

	addi bp, bp, 1          ; bp++
	
	jal r0, loop

end:
	halt
