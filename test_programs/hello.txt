.data
buf:        .byte 11, 'hello world'
out_addr:   .word 0x02

.code
	lui bp, %hi(in_addr)
    ori bp, bp, %lo(in_addr)
    lw bp, bp

	lui s2, %hi(out_addr)
    ori s2, s2, %lo(out_addr)
    lw s2, s2
	
	lb s3, bp
	addi bp, bp, 1
	
loop:
	addi s3, s3, -1			; s3-1 -> s3
	beq s3, r0, end			; if s3 == 0 go to end
	
	lb s4, bp				; mem[bp] -> s4. Символ в s4
	sw s2, s4				; s4 -> mem[s2]. Вывести символ

	addi bp, bp, 1          ; bp++
	
	jal r0, loop

end:
	halt
